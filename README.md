#include<stdio.h>
int main()
{
    int i,j;
    int A[2][3];
    int B[2][3];
    int C[2][3];

printf("Scaning 'A' matrix");
    for(i=0; i<2; i++)
    {
        for(j=0; j<3; j++)
        {
            scanf("%d",&A[i][j]);
        }
        printf("\n");
    }


printf("Scaning 'B' matrix");

    for(i=0; i<2; i++)
    {
        for(j=0; j<3; j++)
        {
            scanf("%d",&B[i][j]);
        }
        printf("\n");

    }


    for(i=0; i<2; i++)
    {
        for(j=0; j<3; j++)
        {
            C[i][j]=A[i][j]+B[i][j];
        }

    }


printf("A + B =");
    for(i=0; i<2; i++)
    {
        for(j=0; j<3; j++)
        {
            printf("  %d ",C[i][j]);
        }
        printf("\n\n");
        printf("       ");

    }
    return 0;
}
